#include <stdio.h>

int main() {
    int n = 5;
    
    
    int arr[n];
    int sum = 0;
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) {
        printf("Element %d: ", i + 1);
        scanf("%d", &arr[i]);
    }
    for (int i = 0; i < n; i++) {
        sum += arr[i];
    }
    printf("The sum of all elements in the array is: %d\n", sum);

}
